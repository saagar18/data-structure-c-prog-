#include<stdio.h>
int main()
{
    int a[20],i,n,flag=0,key,pos;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }
    scanf("%d",&key);
    for(i=0;i<n;i++)
    {
        if(a[i]==key)
        {
            flag++;
            pos=i;
            break;
        }
    }
    if(flag==1)
    {
        printf("the key element found at the %d position",pos);
    }
    else
    {
        printf("the key element not found");
    }
}

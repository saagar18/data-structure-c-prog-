#include <stdio.h>
int main() {
  int n, sum = 0, number;
 printf("Enter the number of elements: ");
  scanf("%d", &n);
  for (int i = 0; i < n; ++i) 
{
     printf("Enter element %d: ", i + 1);
    scanf("%d", &number);

     if (number % 8 == 0) {
      sum += number;
    }
  }
 printf("The sum of all numbers divisible by 8 is: %d\n", sum);
}

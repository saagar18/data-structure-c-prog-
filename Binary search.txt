#include<stdio.h>
int main()
{
    int a[70],i,n,pos,key;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    scanf("%d",&a[i]);
    scanf("%d",&key);
    pos=bin_search(a,n,key);
    if(pos>0)
    {
        printf("element found at the index position %d",pos);
    }
    else
    {
        printf("element not found");
    }
}
int bin_search(int a[],int n,int key)
{
    int first,last,mid;
    first=0;
    last=n-1;
    while(first<=last)
    {
        mid=(first+last)/2;
        if(a[mid]==key)
        {
            return mid+1;
        }
        else if(a[mid]<key)
        {
            first=mid+1;
        }
        else if(a[mid]>key)
        {
            last =mid-1;
        }
    }
    return 0;
}
